1. Basic Vector & Matrix (NO GLM)
#include <cmath>

struct Vec3 {
    float x, y, z;

    Vec3() : x(0), y(0), z(0) {}
    Vec3(float x, float y, float z) : x(x), y(y), z(z) {}
};

struct Mat4 {
    float m[4][4];

    Mat4() {
        for(int i=0;i<4;i++)
            for(int j=0;j<4;j++)
                m[i][j] = (i==j)?1.0f:0.0f;
    }
};

2. Vector Operations (Required for ALL problems)
Vec3 subtract(Vec3 a, Vec3 b) {
    return Vec3(a.x-b.x, a.y-b.y, a.z-b.z);
}

float dot(Vec3 a, Vec3 b) {
    return a.x*b.x + a.y*b.y + a.z*b.z;
}

Vec3 cross(Vec3 a, Vec3 b) {
    return Vec3(
        a.y*b.z - a.z*b.y,
        a.z*b.x - a.x*b.z,
        a.x*b.y - a.y*b.x
    );
}

Vec3 normalize(Vec3 v) {
    float mag = sqrt(dot(v,v));
    return Vec3(v.x/mag, v.y/mag, v.z/mag);
}
3. Matrix Multiplication
Mat4 multiply(Mat4 A, Mat4 B) {
    Mat4 R;

    for(int i=0;i<4;i++) {
        for(int j=0;j<4;j++) {
            R.m[i][j] = 0;
            for(int k=0;k<4;k++) {
                R.m[i][j] += A.m[i][k] * B.m[k][j];
            }
        }
    }
    return R;
}
4. Custom lookAt() (Problem-2)
Matches formula from your sheet:

Zv = N = P0 − Pref
Xv = V × N = U
Yv = N × U

Mat4 myLookAt(Vec3 eye, Vec3 center, Vec3 up)
{
    Vec3 N = normalize(subtract(eye, center));
    Vec3 U = normalize(cross(up, N));
    Vec3 V = cross(N, U);

    Mat4 R;
    R.m[0][0]=U.x; R.m[0][1]=U.y; R.m[0][2]=U.z; R.m[0][3]=0;
    R.m[1][0]=V.x; R.m[1][1]=V.y; R.m[1][2]=V.z; R.m[1][3]=0;
    R.m[2][0]=N.x; R.m[2][1]=N.y; R.m[2][2]=N.z; R.m[2][3]=0;
    R.m[3][0]=0;   R.m[3][1]=0;   R.m[3][2]=0;   R.m[3][3]=1;

    Mat4 T;
    T.m[0][3] = -eye.x;
    T.m[1][3] = -eye.y;
    T.m[2][3] = -eye.z;

    return multiply(R, T);
}
5. Custom Perspective() (Problem-2 / 3)
Mat4 myPerspective(float fov, float aspect, float near, float far)
{
    Mat4 P;

    float t = tan(fov / 2.0f);

    P.m[0][0] = 1.0f / (aspect * t);
    P.m[1][1] = 1.0f / t;
    P.m[2][2] = -(far + near) / (far - near);
    P.m[2][3] = -(2 * far * near) / (far - near);
    P.m[3][2] = -1.0f;
    P.m[3][3] = 0.0f;

    return P;
}
6.Custom Rotate() (Rodrigues Formula)
R=I+(sinθ)K+(1−cosθ)K2
Mat4 myRotate(float theta, Vec3 k)
{
    k = normalize(k);

    float x = k.x, y = k.y, z = k.z;
    float c = cos(theta);
    float s = sin(theta);

    Mat4 R;

    R.m[0][0] = c + x*x*(1-c);
    R.m[0][1] = x*y*(1-c) - z*s;
    R.m[0][2] = x*z*(1-c) + y*s;

    R.m[1][0] = y*x*(1-c) + z*s;
    R.m[1][1] = c + y*y*(1-c);
    R.m[1][2] = y*z*(1-c) - x*s;

    R.m[2][0] = z*x*(1-c) - y*s;
    R.m[2][1] = z*y*(1-c) + x*s;
    R.m[2][2] = c + z*z*(1-c);

    return R;
}

7. Translation + Scaling (Needed for letters)
Mat4 translate(float x, float y, float z)
{
    Mat4 T;
    T.m[0][3] = x;
    T.m[1][3] = y;
    T.m[2][3] = z;
    return T;
}

Mat4 scale(float x, float y, float z)
{
    Mat4 S;
    S.m[0][0] = x;
    S.m[1][1] = y;
    S.m[2][2] = z;
    return S;
}

How to Replace GLM in Your Code

replace glm::mat4 view = basic_camera.createViewMatrix(); with Mat4 view = myLookAt(eye, center, up);
replace glm::perspective(...) with Mat4 projection = myPerspective(fov, aspect, near, far);
replace glm::rotate(...) with Mat4 R = myRotate(angle, axis);
